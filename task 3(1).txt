class Main
{
static String revAlternate(String s,
                            int k, int len)
{
    for (int i = 0; i < s.length();)
    {
    if (i + k > len)
            break;
     s = s.substring(0, i) + new String(new StringBuilder(
            s.substring(i, i + k)).reverse()) +
            s.substring(i + k);
    i += 1 * k;
    }
    return s;
}
public static void main(String[] args)
{
    String s = "important";
    int len = s.length();
    int k = 2;
    System.out.println(revAlternate(s, k, len));
}
}
